-- criar tabela pessoa
create table pessoa(
id pessoa int primary key,
nome varchar(45),
peso float,
altura double,
salario decimal(10,2)
);

--tipagem números decimais
-- Float - 7 dígitos (32 bits)
-- Double - 15 digitos (64 bits)
-- Decimal (t, após virgula) 28 digitos (128 digitos)
